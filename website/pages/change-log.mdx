import { useContext } from "react";
import ReleasesRenderer from "../components/release";

export const getStaticProps = ({ params }) => {
  return (
    fetch("https://api.github.com/repos/weiran-zsd/dts-cli/releases")
      .then((res) => res.json())
      // we keep the most recent 3 releases here
      .then((releases) => ({ props: { ssg: releases.slice(0, 10) } }))
  );
};

# Change Log

Please visit the [DTS-cli release page](https://github.com/formium/dts-cli/releases) for all historical releases.

<ReleasesRenderer releases={props.ssg} />
